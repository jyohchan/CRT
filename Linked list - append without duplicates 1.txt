class Node:
    def __init__(self,val):
        self.val=val
        self.next=None
class Single:
    def __init__(self):
        self.head=self.tail=None
    def create(self,Val):
        n=Node(Val)
        if self.head==None:
            self.head=self.tail=n
        elif self.find(Val) is None:
            self.tail.next=n
            self.tail=n
        else:
            pass
    def find(self,data):
        temp=self.head
        while temp!=None:
            if temp.val==data:
                return temp
            temp=temp.next
        return None
    def show(self):
        temp=self.head
        while temp!=None:
            print(temp.val)
            temp=temp.next
